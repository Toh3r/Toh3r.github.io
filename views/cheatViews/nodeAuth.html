BcryptJS

npm page
Uses hash and salt 


                const bcrypt = require('bcryptjs');

                const hashFunction = async () => {
                  const password = 'hello123';
                  // passing pword + number of hashing rounds
                  // Bcrypt returns a promise
                  const hashed = await bcrypt.hash(password, 8);

                  console.log(hashed);

                  // Check if paswwords match
                  const isMatch = await bcrypt.compare('hello123', hashed);
                  // Returns a boolean
                  console.log(isMatch);
                };
                
                hashFunction();

Json Web Tokens (JWT) npm
Medium article on JWT 


                const jwt = require('jsonwebtoken');

                const jwtFunction = async () => {
                  // Create json token, passing through unique id for user, token signature ('secret') and customizable options
                  const token = jwt.sign({ _id: '123454321' }, 'random characters', {
                    expiresIn: '7 days'
                  });
                
                  // Verify token, passing token and secret
                  jwt.verify(token, 'random characters');
                };
                
                jwtFunction();
