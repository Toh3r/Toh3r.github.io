SSH Keys

            // Check for ssh keys -> id_rsa id_rsa.pub files
            ls -a -l ~/.ssh 

            // Create ssh key -> Creates id_rsa/pub files
            ssh-keygen -t rsa -b 4096 -C 'email'

            // Start ssh agent
            eval $(ssh-agent -s)

            // Register file
            ssh-add ~/.ssh/id_rsa

            // Print public rsa key to console
            cat ~/.ssh/id_rsa.pub

            // Test connection to github servers
            ssh -T git@github.com
